// a utility for timing out a Promise
function timeoutPromise(delay) {
	return new Promise( function(resolve,reject){
		setTimeout( function(){
			reject( "Timeout!" );
		}, delay );
	} );
}

Promise.race( [timeoutPromise(3000), fetch('http://www.kuaizhan.com')] )
.then( function(msg){
    console.log(msg);
